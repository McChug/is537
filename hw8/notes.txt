class Solution:
    def isPalindrome(self, head: Optional[ListNode]) -> bool:
        stack = []
        cur = head

        if head.next == None:
            return True

        while cur:
            if len(stack) > 0 and stack[-1] == cur.val:
                stack.pop()
            else:
                stack.append(cur.val)

            cur = cur.next

        if len(stack) == 0:
            return True

        return False
